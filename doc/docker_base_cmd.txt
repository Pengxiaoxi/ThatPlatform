# 生成镜像
docker build -t tpf_docker .


# 查看镜像
docker images


# 使用镜像创建容器并运行
docker run --name=tpf_docker -p 8000:8000 -d tpf_docker


# 查看运行容器
docker ps -a

# Stop ocker容器
docker stop 1d52b2b722ad

# Remove Docker容器
docker rm 1d52b2b722ad

# Start Docker 容器
docker start 1d52b2b722ad
